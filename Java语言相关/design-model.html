class Singleton1 {
    private Singleton1() {

    }

    /*дҝқиҜҒдҝ®ж”№еҗҺеӨҡзәҝзЁӢеҸҜд»Ҙи§Ғ*/
    private volatile static Singleton1 singleton1 = null;

    /*еҜ№еӨ–йғЁиҺ·еҸ–еҜ№иұЎжҸҗдҫӣеҗҢжӯҘзҡ„ж–№жі•*/

    public synchronized static Singleton1 getInstance() {
        if (singleton1 == null) {
            singleton1 = new Singleton1();
            return singleton1;
        }
        return singleton1;
    }

йҖӮй…ҚеҷЁи®ҫи®ЎжЁЎејҸпјҡ дёҖз§Қи®©дёӨдёӘжҺҘеҸЈдә’зӣёе…је®№зҡ„дёҖз§Қи®ҫи®ЎжЁЎејҸпјҢеҸҜд»ҘеҸӮиҖғз”өжәҗйҖӮй…ҚеҷЁпјҢжүӢжңәжғіиҰҒе……з”өдҪҶжҳҜпјҢеҝ…йЎ»дҪҝз”Ёз”өжәҗйҖӮй…ҚеҷЁе°ҶжҺҘеҸЈиҪ¬жҚўжҲҗжүӢжңәиғҪдҪҝз”Ёзҡ„з”өеҺӢе’Ңз”өжөҒгҖӮ

з”ЁGenericServletдёҫдҫӢ иҝҷдёӘзұ»жҳҜдёҖдёӘз®ҖеҚ•дҪҝз”ЁServletзҡ„йҖӮй…ҚеҷЁпјҢиҝҷдёӘзұ»еҸҜд»Ҙж»Ўи¶іеҸӘйҮҚеҶҷдёҖдёӘж–№жі•е°ұиғҪ е®һзҺ°дёҖдёӘServletгҖӮ GenericServletеҶ…йғЁе®һзҺ°дәҶServletзҡ„дёҖдәӣж–№жі•пјҢе№¶дё”е·§еҰҷзҡ„и®ҫи®ЎжҡҙйңІеҮәжқҘзҡ„ж–№жі•пјҢ и®©дҪҝз”ЁеҸҳеҫ—жӣҙеҠ з®ҖеҚ•гҖӮ з»§жүҝжӯӨжҠҪиұЎзұ»еҸӘйңҖиҰҒйҮҚеҶҷserviceж–№жі•пјҢеҰӮжһңжғіиҰҒйҮҚеҶҷinitеҸҜд»ҘйҮҚеҶҷGenericServletзҡ„ж— еҸӮinitжһ„йҖ ж–№жі•гҖӮиҖҢдёҚйңҖиҰҒйҮҚеҶҷServletзҡ„жңүеҸӮжһ„йҖ ж–№жі•гҖӮ

иў«и§ӮеҜҹиҖ…еҰӮдёӢ

java
import java.util.ArrayList;
import java.util.List;

/**
 * иў«и§ӮеҜҹзҡ„иҖҒдәә
 */
public class OldMan implements ToBeObserver {
    /**
     * и§ӮеҜҹиҖ…йӣҶеҗҲ
     */
    private List<Observer> observers;

    /**еўһеҠ и§ӮеҜҹиҖ…*/
    @Override
    public void addObserver(Observer observer) {
        if (null == observers) {
            observers = new ArrayList<>();
            observers.add(observer);
        } else {
            observers.add(observer);
        }
    }

    /**еҲ йҷӨи§ӮеҜҹиҖ…*/
    @Override
    public void removeObserver(Observer observer) {
        observers.remove(observer);
    }
    /**йҖҡзҹҘи§ӮеҜҹиҖ…*/
    @Override
    public void notifyObserver(String message) {
        observers.forEach(observer -> observer.handleNotify(message));
    }
}
